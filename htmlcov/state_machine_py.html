131def parse_duration(duration): 

132 # ["ms", "m", "s", "h", "d"]; 

133 if duration == 'rand:1d,4d': 

134 return 86400 # 1d 

135 elif "ms" in duration: 

136 return float(duration[:-2]) / 1000 

137 elif "s" in duration: 

138 return float(duration[:-1]) 

139 elif "m" in duration: 

140 return float(duration[:-1]) * 60 

141 elif "h" in duration: 

142 return float(duration[:-1]) * 3600 

143 elif "d" in duration: 

144 return float(duration[:-1]) * 86400 

145 else: 

146 return float(duration)
